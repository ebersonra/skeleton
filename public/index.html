🧪 Exemplos de Testes

Unit Test 
// src/tests/unit/exampleService.test.js
const exampleService = require('../../services/exampleService');

describe('exampleService', () => {
  test('should return service info', () => {
    const result = exampleService.getInfo();
    expect(result).toBeDefined();
    expect(result.status).toBe('OK');
    expect(result.version).toBe('1.0.0');
  });
});
Integration Test 
// src/tests/integration/health.integration.test.js
const healthController = require('../../controllers/healthController');
const ExampleModel = require('../../models/exampleModel');

const mockResponse = () => ({
  json: jest.fn().mockReturnThis(),
  status: jest.fn().mockReturnThis()
});

test('should integrate controller -> service -> model', () => {
  const req = {};
  const res = mockResponse();
  const originalCount = ExampleModel.count;
  ExampleModel.count = jest.fn().mockReturnValue(5);

  healthController(req, res);

  expect(ExampleModel.count).toHaveBeenCalled();
  expect(res.json).toHaveBeenCalledWith(
    expect.objectContaining({
      status: 'OK',
      exampleCount: 5
    })
  );
  
  ExampleModel.count = originalCount;
});
E2E Test 
// src/tests/e2e/api.test.js
const request = require('supertest');
const Server = require('../../server');

describe('E2E Tests - API Endpoints', () => {
  let app, server;

  beforeAll(() => {
    server = new Server();
    app = server.app;
  });

  test('should return health status', async () => {
    const response = await request(app)
      .get('/api/health')
      .expect(200);

    expect(response.body).toHaveProperty('status');
    expect(response.body.status).toBe('OK');
  });
});

⚡ Comandos Essenciais

🧪 Testes

npm test Executa todos os testes
npm run test:unit Apenas testes unitários
npm run test:e2e Testes end-to-end
npm run test:watch Modo watch para desenvolvimento

🔍 Qualidade de Código

npx eslint . Análise de código completa
npx eslint --fix . Corrige problemas automaticamente

🚀 Desenvolvimento

npm start Inicia servidor (porta 4000)
netlify dev Preview local com Netlify (porta 8888)

⚡ Setup Rápido

1

Clone & Install

git clone https://github.com/ebersonra/skeleton.git
cd skeleton
npm install
2

Configurar Ambiente

# Criar arquivo .env
PORT=4000
CORS_ORIGIN=http://localhost:8888
3

Rodar & Testar

npm start        # Servidor
npm test         # Testes
npx eslint .     # Lint
